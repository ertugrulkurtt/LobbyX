rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Allow authenticated users to upload and read profile photos
    match /profile-photos/{allPaths=**} {
      allow read, write: if request.auth != null;
    }
    
    // Allow users to manage their own profile photos
    match /users/{userId}/profile-photo/{allPaths=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow public profile photos for all authenticated users
    match /public/profile-photos/{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null 
        && request.resource.size < 5 * 1024 * 1024 
        && request.resource.contentType.matches('image/.*');
    }
    
    // Allow authenticated users to upload to root (temporary fix)
    match /{filename} {
      allow read, write: if request.auth != null 
        && request.resource.size < 5 * 1024 * 1024 
        && request.resource.contentType.matches('image/.*');
    }
    
    // Fallback rule for any file - very permissive for testing
    match /{allPaths=**} {
      allow read, write: if request.auth != null;
    }
  }
}
