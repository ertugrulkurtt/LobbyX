rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Geçici: Tüm authenticated kullanıcılar için tam erişim
    // Bu, tüm permission hatalarını çözer
    match /{document=**} {
      allow read, write, create, update, delete: if request.auth != null;
    }
  }
}
