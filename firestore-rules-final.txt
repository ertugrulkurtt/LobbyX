rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Users collection - allow read for all authenticated users, write for own data
    match /users/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Friend requests - allow read/write for sender and receiver
    match /friendRequests/{requestId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
      allow create: if request.auth != null;
    }

    // Conversations - allow read/write for participants
    match /conversations/{conversationId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
      allow create: if request.auth != null;
    }

    // Messages - allow read/write for authenticated users
    match /conversations/{conversationId}/messages/{messageId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
      allow create: if request.auth != null;
    }

    // Servers - allow read/write for authenticated users
    match /servers/{serverId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
      allow create: if request.auth != null;
    }

    // Server members
    match /servers/{serverId}/members/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
      allow create: if request.auth != null;
    }

    // User statistics
    match /userStats/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId;
    }

    // Daily activity
    match /dailyActivity/{activityId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
      allow create: if request.auth != null;
    }

    // Message statistics
    match /messageStats/{messageStatsId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
      allow create: if request.auth != null;
    }

    // XP logs
    match /xpLogs/{logId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
      allow create: if request.auth != null;
    }

    // Achievement progress
    match /achievementProgress/{progressId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
      allow create: if request.auth != null;
    }

    // Leaderboard
    match /leaderboard/{document=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }

    // Groups
    match /groups/{groupId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
      allow create: if request.auth != null;
    }

    // Server invites
    match /serverInvites/{inviteId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
      allow create: if request.auth != null;
    }

    // Notifications
    match /notifications/{userId}/userNotifications/{notificationId} {
      allow read: if request.auth != null && request.auth.uid == userId;
      allow write: if request.auth != null && request.auth.uid == userId;
    }

    // Allow all other authenticated operations (temporary)
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
